Example

# main.tf
resource "azurerm_resource_group" "rg" {
  location = "uksouth"
  name     = "rg-example-1234"
}

resource "azurerm_virtual_network" "vnet" {
  address_space       = ["10.254.0.0/16"]
  location            = azurerm_resource_group.rg.location
  name                = "vnet-example-1234"
  resource_group_name = azurerm_resource_group.rg.name
}

resource "azurerm_subnet" "snet-frontend" {
  address_prefixes     = ["10.254.0.0/24"]
  name                 = "snet-frontend-example-1234"
  resource_group_name  = azurerm_resource_group.rg.name
  virtual_network_name = azurerm_virtual_network.vnet.name
}

resource "azurerm_subnet" "snet-backend" {
  address_prefixes     = ["10.254.2.0/24"]
  name                 = "snet-backend-example-1234"
  resource_group_name  = azurerm_resource_group.rg.name
  virtual_network_name = azurerm_virtual_network.vnet.name
}

resource "azurerm_public_ip" "pip" {
  allocation_method   = "Dynamic"
  location            = azurerm_resource_group.rg.location
  name                = "pip-example-1234"
  resource_group_name = azurerm_resource_group.rg.name
}

locals {
  backend_address_pool_name      = "${azurerm_virtual_network.vnet.name}-bapn"
  frontend_port_name             = "${azurerm_virtual_network.vnet.name}-fpn"
  frontend_ip_configuration_name = "${azurerm_virtual_network.vnet.name}-fipcn"
  http_setting_name              = "${azurerm_virtual_network.vnet.name}-hsn"
  listener_name                  = "${azurerm_virtual_network.vnet.name}-ln"
  request_routing_rule_name      = "${azurerm_virtual_network.vnet.name}-rrrn"
}

module "azurerm_application_gateway" {
  source = "../../"

  backend_address_pools = [{
    name = local.backend_address_pool_name
  }]
  backend_http_settings = [{
    cookie_based_affinity = "Disabled"
    name                  = local.http_setting_name
    path                  = "/path1/"
    port                  = 80
    protocol              = "Http"
    request_timeout       = 60
  }]
  frontend_ip_configurations = [{
    name                 = local.frontend_ip_configuration_name
    public_ip_address_id = azurerm_public_ip.pip.id
  }]
  frontend_ports = [{
    name = local.frontend_port_name
    port = 80
  }]
  gateway_ip_configurations = [{
    name      = "config1"
    subnet_id = azurerm_subnet.snet-frontend.id
  }]
  http_listeners = [{
    frontend_ip_configuration_name = local.frontend_ip_configuration_name
    frontend_port_name             = local.frontend_port_name
    name                           = local.listener_name
    protocol                       = "Http"
  }]
  location = "uksouth"
  name     = "example"
  request_routing_rules = [{
    backend_address_pool_name  = local.backend_address_pool_name
    backend_http_settings_name = local.http_setting_name
    http_listener_name         = local.listener_name
    name                       = local.request_routing_rule_name
    rule_type                  = "Basic"
  }]
